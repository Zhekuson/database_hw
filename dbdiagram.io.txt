Table users as U {
  id int [pk, increment]
  name nvarchar(30) [not null]
  surname nvarchar(30) [not null]
  middle_name nvarchar(30)
  birth_date datetime
  class int
  is_graduate bool
  email nvarchar(55)
  photo varbinary
}

Table olimpiads as O {
  id int [pk, increment]
  name nvarchar(100)
  logo varbinary
  level int
  university_id int
}

Table olympiad_subjects as OSUBJ { 
  olympiad_id int
  subject subject
  Indexes {
    (olympiad_id, subject) [pk]
  }
}

Table olimpiad_rounds as ORND {
  id int
  olympiad_id int
  is_online bool
  registration_date datetime
  registration_end_date datetime
  start_date datetime
  end_date datetime
  place nvarchar(100)
  is_final bit
  Indexes {
    (id, olympiad_id) [pk]
  }
}

Table olympiad_criteria as OCRIT { 
  round_id int
  diploma_level nvarchar(50)
  min_score int
}

Table olympiad_bonuses as OBONUS { 
  olympiad_id int
  university_id int
  
  diploma_level nvarchar(50)
  //BVI, +n points
  bonus_type nvarchar(50)
  bonus_points int
  Indexes {
    (university_id, olympiad_id) [pk]
  }
}

Table user_olimpiad as UINO { 
  user_id int
  olympiad_id int
  result varchar(20)
  score int
  Indexes {
    (user_id, olympiad_id) [pk]
  }
}

Table friends as F {
  first_id int
  second_id int
  start_date datetime
  Indexes {
    (first_id, second_id) [pk]
  }
}

Table staff as STAFF {
  id int [pk, increment]
  name nvarchar(30) [not null]
  surname nvarchar(30) [not null]
  middle_name nvarchar(30)
  birth_date datetime
  email nvarchar(50)
  photo varbinary
  is_tutor bool
  position nvarchar(50)
  price int
}

Table staff_university as STUNV {
  staff_id int
  university_id int
  Indexes {
    (staff_id, university_id) [pk]
  }
}

Table staff_olympiad as STOLYMP { 
  olympiad_id int
  staff_id int
  Indexes {
    (staff_id, olympiad_id) [pk]
  }
}

Table staff_subject as STSUBJ {
  staff_id int
  subject subject
  Indexes {
    (staff_id, subject) [pk]
  }
}

Table staff_student as STSTUD { 
  staff_id int
  user_id int
  is_active bool
  start_date datetime
  end_date datetime
  Indexes {
    (staff_id, user_id) [pk]
  }
}

Table university as UNV {
  id int [pk, increment]
  full_name narchar(255)
  address nvarchar(255)
  phone varchar(20)
  email nvarchar(50)
}

Table course_olympiad as CRSOLYMP {
  id int [pk, increment]
  olympiad_id int
  staff_id int
  subject subject
  school_class int
  url nvarchar(255)
}

Table user_round as UR {
  id int [pk, increment]
  user_id int
  round_id int
  registration_date datetime
  //Approved, processing
  status nvarchar(20)
}

Table registration_data_unit as RDU { 
  registration_id int 
  key nvarchar(255)
  value nvarchar(255)
  Indexes {
    (registration_id, key) [pk]
  }
}

Table required_registration_rata as RRD {
  registration_id int
  key nvarchar(255)
  value nvarchar(255)
  Indexes {
    (registration_id, key) [pk]
  }
}
